Config.json

Este archivo se ubica en styflux/config , yo podría llamar a este archivo el cerebrito del styflux ya que aquí se declaran las variables entre otras configuraciones para administrar a tu necesidad el Styflux.

1. "Compile" aquí indicamos donde se compilara(renderizara) nuestros estilos:

"compile":{
	"ext":".styl",
	"cmd":"stylus -c",
	"cmdr":"-o",
	"route":"../css" <!-- esto varía segun nuestra ubicación de carpeta contenedora de css -->
}

2. "Layout" contiene la suma de los estilos que estan en styflux/view-helpers:

"layouts":{

"active":true,
"path":"./layouts",

"compile":{
	"layout.styl":{
"helpers":{
"fonts":"fonts.styl",
"content":"content.styl",
"reset":"reset.styl",
"grids":"grids.styl",
"buttons":"buttons.styl",
"forms":"forms.styl",
"table":"table.styl",
"plugins":"plugins.styl",
"resources":["resources.styl"],
"hacks":"[@ult]hacks.styl"
}
}
}
